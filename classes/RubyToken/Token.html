  def open_tag?
    return false if @name.nil? || get_props.nil?
    a = (get_props[1] == RubyToken::EXPR_BEG) &&
          self.class.to_s !~ /_MOD/  && # ignore onliner if, unless, etc...
          !FAKIES.member?(self.class)  
    a 
  end
